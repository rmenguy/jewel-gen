       ```
       handleExecuteStack: async () => {
         const session = structuredClone(store.stackSession) with chatSession nulled
         initializeStepStates(mutableSession)
         store.updateStackSession({ status: 'executing', stepStates: mutableSession.stepStates })
         try {
           await executeStackPlan(mutableSession, onStepUpdate callback that syncs to store)
           compactSnapshots(mutableSession)
           store.updateStackSession(final state from mutableSession)
         } catch {
           store.updateStackSession({ status: 'planning' })
         }
       }
       ```

       **Follow-up handler** (per research follow-up wiring):

       ```
       handleFollowUp: async (prompt) => {
         const mutableSession = { ...store.stackSession }
         if (!mutableSession.chatSession) initFollowUpSession(mutableSession)
         store.updateStackSession({ status: 'follow-up' })
         const newImage = await sendFollowUpEdit(mutableSession, prompt)
         store.updateStackSession({ currentImage: newImage, followUpHistory: mutableSession.followUpHistory })
       }
       ```
