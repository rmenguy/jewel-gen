  'collier': 'collarbone',

  'earring': 'ear-lobe',

export function autoAssignZone(category: string): TargetZone {
  const lower = category.toLowerCase();
  for (const [key, zone] of Object.entries(CATEGORY_TO_ZONE)) {
    if (lower.includes(key)) return zone;
  }
  return 'collarbone'; // safe default for unrecognized categories
}

export function getZonePlacementPrompt(zone: TargetZone): string {

Add these new fields to the `ProductionStore` interface:

```typescript
// Stack session state (STATE-01, STATE-02)
stackSession: ProductionStackSession | null;

// Stack session actions
createStackSession: (baseImage: string, aspectRatio: string, imageSize: string) => ProductionStackSession;
updateStackSession: (updates: Partial<ProductionStackSession>) => void;
addLayerToStack: (layer: StackLayer) => void;
removeLayerFromStack: (layerId: string) => void;
reorderStackLayers: (layerIds: string[]) => void;
updateStepState: (stepIndex: number, updates: Partial<StepState>) => void;
resetStackSession: () => void;
```

Implementation details for each action:

1. `createStackSession`: Creates a new `ProductionStackSession` with `id: crypto.randomUUID()`, the provided `baseImage`, `aspectRatio`, `imageSize`, empty `layers: []`, empty `stepStates: []`, `currentImage: null`, `chatSession: null`, empty `followUpHistory: []`, `status: 'planning'`, `createdAt: Date.now()`. Sets it as `stackSession` and returns it.

2. `updateStackSession`: Merges partial updates into the current `stackSession`. If `stackSession` is null, no-op.

3. `addLayerToStack`: Appends a `StackLayer` to `stackSession.layers`. If `stackSession` is null, no-op.

4. `removeLayerFromStack`: Removes the layer with matching `layerId` from `stackSession.layers` and re-assigns `ordinal` values (0-indexed) to remaining layers.

5. `reorderStackLayers`: Takes an array of `layerId` strings in desired order. Rebuilds `stackSession.layers` in that order, re-assigning `ordinal` values.

6. `updateStepState`: Updates `stackSession.stepStates[stepIndex]` with partial updates. If session or index invalid, no-op.

7. `resetStackSession`: Sets `stackSession` to `null`.

Add the new imports at the top: `ProductionStackSession, StackLayer, StepState` from `'../types'`.

CRITICAL: Use `get()` inside actions (not hook closures) to avoid stale state in async operations (per Pitfall 5 from research).
